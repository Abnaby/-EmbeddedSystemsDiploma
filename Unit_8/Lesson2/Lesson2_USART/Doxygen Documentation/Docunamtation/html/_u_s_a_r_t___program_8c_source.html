73void
74MCAL_USART_init(UART_cnfg* ptr_UsartConfig)
75{
76 /* USART STATE */
77 USART_UCSRB &= ~(0b11 << 3); // Clear Position at first
78 USART_UCSRB |= (((ptr_UsartConfig)->state) << 3) ; // Set USART State
79 /* Frame Configuration */
80 /* GET_MSB == 2 YES SET BIT TO 9TH DATA NO CLEAR BIT */
81 (((ptr_UsartConfig->databits >> 2) & 0b001) == 1) ? (USART_UCSRB |= (UCSRB_UCSZ2)) : (USART_UCSRB &= ~(UCSRB_UCSZ2));
82
83 USART_UCSRC = (UCSRC_URSEL) // SET FRAME FORMAT
84 | ((ptr_UsartConfig->databits & 0b011) << 1 ) // get bit num 2, 3 in 0bxxx then shift it to bit number 1 in UCSRC
85 | (ptr_UsartConfig->parity) // Set Parity Bits
86 | (ptr_UsartConfig->stopbits) ; // Set Number of Stop Bits
87 /* SET BAUDRATE */
88 USART_UBRRL = (u8)BAUD_PRESCALE(ptr_UsartConfig->baudrate) ;
89 USART_UBRRH = (u8)BAUD_PRESCALE(ptr_UsartConfig->baudrate) >> 8;
90}

104void
105MCAL_USART_TxSynch(u32 copy_u32Data)
106{
107 /* CHECK READY TO SEND OR NOT */
108 WAIT_TO_TRANSMIT(); /* Wait for empty transmit buffer*/
109 if(Global_uartcfg->databits > _8BIT )
110 {
111 u32 data = copy_u32Data & 0x1FF ;
112 /*
113 TXB8 is the ninth data bit in the character to be transmitted when operating with serial frames with nine data
114 bits. Must be written before writing the low bits to UDR.
115 */
116 /* Copy 9th bit to TXB8 */
117 (data = data >> 8) ? (USART_UCSRB |= UCSRB_TXB8) : (USART_UCSRB &= ~UCSRB_TXB8) ;;
118 /* Put data into buffer, sends the data */
119 USART_UDR = copy_u32Data & 0xFF0;
120 }
121 else
122 {
123 u8 *LOC_u8DataBuffer = (u8*) &copy_u32Data ;
124 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[0]);
125 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[1]);
126 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[2]);
127 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[3]);
128 }
129}

138void
139MCAL_USART_RxSynch(u32 *ptr_u32ReceivedData)
140{
141 /* WAIT UNTIL RECIEVE COMPLATE */
142 WAIT_TO_RECIEVE(); /* Wait till data is received */
143 if(Global_uartcfg->databits > _8BIT )
144 {
145 /* Get status and 9th bit, then data */
146 /* from buffer */
147 u8 resh = USART_UCSRB;
148 u8 resl = USART_UDR;
149 /* Filter the 9th bit, then return */
150 resh = (resh >> 1) & 0x01;
151 *ptr_u32ReceivedData = ((resh << 8) | resl);
152 }
153 else
154 {
155 u32 LOC_u32RecievedData = 0 ;
156 u8* LOC_u8PtrToData = (u8 *)&LOC_u32RecievedData ;
157 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[0]);
158 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[1]);
159 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[2]);
160 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[3]);
161 *ptr_u32ReceivedData = LOC_u32RecievedData;
